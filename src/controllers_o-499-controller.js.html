/**
 * @access public
 * @author @victorinknov
 * @since 19/02/2024 11:56:20
 * @description (Controller) -  Componente de Ferramentas Globais Unificadas (observações, documentos complementares)
 **/

import { C499 as service } from "../services/o-499-service.js";
import { componentOrchestrator, awaitComponentLoad, taskRules, importComponent } from "../core/componentController.js";
import * as debug from "../../assets/javascripts/console-debug.js";
import { startLoading, endLoading } from "../../assets/javascripts/normalize.js";

/**
 * @constructs C499
 * @module C499
 * @constant C499
 * @type {object}
 * @exports C499
 **/

const C499 = {
    async init(params) {
        const componentService = service.schema();
        if (params.visible) {
            if (!params.refresh) debug.init("O-499");
            else debug.refresh("O-499");
            const componentData = await service.data(params);
            componentOrchestrator(params, componentService, componentData);
            await awaitComponentLoad(componentService.component.target);
            await this.loadRules(componentData.data);
        } else {
            document.getElementById(componentService.component.target).innerHTML = "";
        }
        return { visible: params.visible, loadedSuccessfully: true };
    },
    async refresh(params) {
        await this.init({ ...params, refresh: true });
    },
    async loadRules(data) {
        const { SCREEN_HASH } = sessionStorage;
        sessionStorage.setItem("MC499Data", data);
        const dropdownButton = document.getElementById("toolbox-dropdown-btn");
        const dropdownContainer = document.querySelector(".toolbox-dropdown-container");
        const body = document.querySelector("body");

        dropdownButton.addEventListener("click", function () {
            const isActive = dropdownContainer.classList.toggle("active-v");
            const btn3 = $(".toolbox-component-item:nth-of-type(3)");
            const btn2 = $(".toolbox-component-item:nth-of-type(2)");
            const btn1 = $(".toolbox-component-item:nth-of-type(1)");

            const buttonsAnimated = [
                {
                    key: "btn1",
                    btnEl: btn1,
                },
                {
                    key: "btn2",
                    btnEl: btn2,
                },
                {
                    key: "btn3",
                    btnEl: btn3,
                },
            ];
            const inn = isActive ? "in" : "out";
            const out = isActive ? "out" : "in";

            buttonsAnimated.forEach((btn) => {
                btn.btnEl.removeClass(`animate-${out}-${btn.key}`);
                btn.btnEl.addClass(`animate-${inn}-${btn.key}`);
            });
            $("[class*='btn3']").addClass("d-none");
            if (["P1TL15", "P2TL09"].includes(SCREEN_HASH)) {
                $("[class*='btn1']").addClass("d-none");
            }
        });

        window.addEventListener("click", function (event) {
            if (!dropdownContainer.contains(event.target) && dropdownContainer.classList.contains("active-v")) {
                dropdownContainer.classList.remove("active-v");
            }
        });
        body.style.marginBottom = "5rem";
        await this.verificarStatusComponentes();
    },
    async verificarStatusComponentes() {
        const C166 = await importComponent("o-166-controller.js");
        const MC166 = Object.create(Object.assign(C166));
        const MC166Data = await MC166.init({ visible: "data" });

        const C179 = await importComponent("o-179-controller.js");
        const MC179 = Object.create(Object.assign(C179));
        const MC179Data = await MC179.init({ visible: "data" });

        if (MC166Data.componentData.data.length || MC179Data.componentData.data.length) {
            await this.componentesObservacoes(true);
        }

        const CG904 = await importComponent("o-ged-904-controller.js");
        const MCG904 = Object.create(Object.assign(CG904));
        const MCG904Data = await MCG904.init({ visible: "data" });

        if (MCG904Data.componentData.data.length) {
            await this.componentesGED(true);
        }
    },
    async componentesObservacoes(isVerifying) {
        const { SCREEN_HASH } = sessionStorage;
        if (!isVerifying) startLoading();
        const C161 = await importComponent("o-161-controller.js");
        const MC161 = Object.create(Object.assign(C161));
        window.MC161 = MC161;
        await MC161.init({ visible: true });

        if (!["P1TL15", "P2TL09"].includes(SCREEN_HASH)) {
            const C166 = await importComponent("o-166-controller.js");
            const MC166 = Object.create(Object.assign(C166));
            window.MC166 = MC166;
            await MC166.init({ visible: true });
        }

        const C179 = await importComponent("o-179-controller.js");
        const MC179 = Object.create(Object.assign(C179));
        window.MC179 = MC179;
        await MC179.init({ visible: true });
        if (!isVerifying) endLoading();
    },
    async componentesGED(isVerifying) {
        if (!isVerifying) startLoading();
        const CG904 = await importComponent("o-ged-904-controller.js");
        const MCG904 = Object.create(Object.assign(CG904));
        window.MCG904 = MCG904;
        await MCG904.init({ visible: true });

        const CG930 = await importComponent("o-ged-930-controller.js");
        const MCG930 = Object.create(Object.assign(CG930));
        window.MCG930 = MCG930;
        await MCG930.init({ visible: true });

        if (!window.MC907) {
            const C907 = await importComponent("o-907-controller.js");
            const MC907 = Object.create(Object.assign(C907));
            window.MC907 = MC907;
            await MC907.init({ visible: true });
        } else {
            document.getElementById("x-tc499")?.querySelector("#modal-visualizar-arquivo-pdf")?.remove();
        }

        if (!isVerifying) endLoading();
    },
    async componenteAcoes() {
        const C1010 = await importComponent("o-1010-controller.js");
        const MC1010 = Object.create(Object.assign(C1010));
        window.MC1010 = MC1010;
        await MC1010.init({ visible: true });
    },
};

export default C499;
