/**
 *
 * @param {Object} params The parameters that componentOrchestrator will use to build the component.
 * @param {Object} schema The details of the component, like sources and more.
 * @param {Object} data The data that will be appended to the component. It can be a mock data or data from API.
 * @function componentOrchestrator
 * @async
 * @throws {Error} - Throws an error if component construction encounters an issue.
 * @description This function is an orchestrator to mount components.
 */
export async function componentOrchestrator(params, schema, data) {
    /**
     * organismSource and moleculeSource are the paths to the
        file template to build the component. organism must be 
        the main content of the component otherwise the molecule
        must be the repeated part of the component. always think
        in a table: the organism is the table in group with 
        table headers, table footers... while the molecule is
        a simple table row.
        --
        in case of the molecule component doesn't provide any data
        and it's just the template, in the component's service file
        return something like this:
        return [''];
     */
    const organismSource = originCDN + "templates/organisms/";
    const moleculeSource = originCDN + "templates/molecules/";
    // 00: request organism template

    const { viewType } = params;

    const viewTypeModel = viewType ? viewType + "/" : "";

    $.get(organismSource + viewTypeModel + schema.component.organism + ".html", (organism) => {
        // 01: if there's no target defined in params, take it from component schema
        if (!params.target) params.target = schema.component.target;
        // 02: by default, always delete all childrens of the target
        document.getElementById(params.target).innerHTML = "";
        // 03: append core data defined into the component
        $("#" + params.target).append(buildComponent(organism, data?.component));
        // 04: request molecule template
        const dataSources = Object.keys(data).filter((key) => key.toLowerCase().includes("data"));
        dataSources.map((source, idx) => {
            const moleculeOrigin = schema.component.molecule + (idx ? "-origin-" + idx : "") + ".html";
            const moleculeHydrated = "#molecule-hydrated" + (idx ? "-origin-" + idx : "");
            $.get(moleculeSource + moleculeOrigin, (molecule) => {
                // 05: data must be always converted to an array, it will make easier append into the component
                const dataConvertedToArray = Array.isArray(data[source]) ? data[source] : $.isEmptyObject(data[source]) ? [] : [data[source]];
                // 06: if there is no data, return append something to the component to awaitComponentLoad resolve
                if (!dataConvertedToArray.length || dataConvertedToArray == "[]") {
                    document.getElementById(params.target).querySelector(moleculeHydrated).innerHTML = '<div class="empty"></div>';
                }
                // 07: otherwise, for each element of the array, build the component
                else {
                    const buildedMolecules = dataConvertedToArray
                        .map((object) => {
                            return buildComponent(molecule, object);
                        })
                        .join("");
                    // 08: then, after all, append builded molecules into the molecule default target.
                    try {
                        document.getElementById(params.target).querySelector(moleculeHydrated).innerHTML = buildedMolecules;
                    } catch (error) {
                        console.error(
                            `\n Não foi encontrado nenhuma molecula a ser hidratada em ${schema.component.organism}. \n O alvo principal era: ${params.target}. \n A quantidade de itens dentro do nível "data" da service é: ${data[source].length}.\n O esquema de fallback do componente foi definido em ${schema.component.fallbackMoleculeFailure}`
                        );
                        console.table(params);
                        console.table(schema.component);
                    }
                }
            });
        });
    });
}

/**
 * 
 * @param {String} template 
 * @param {Object} object 
 * @returns {String}
 * @function buildComponent
 * @victorinknov
 * @description
    This function returns a modified version of the template by replacing specific placeholders 
    with corresponding values from the object. Inside the loop, the template is modified
    using the replace method. It searches for a placeholder pattern {{atom_data:key}} in 
    the template and replaces it with the corresponding value from the object. After iterating
    over all key-value pairs, the template may still contain unresolved placeholders that weren't
    found in the object. To remove these placeholders, another replace method is used with the regex
    /{{atom_data:[^}]+}}/g. The placeholders are replaced with an empty string, effectively removing 
    them from the template.
 */
function buildComponent(template, object = {}) {
    for (const [key, value] of Object.entries(object)) {
        template = template.replace(new RegExp(`{{atom_data:${key}}}`, "g"), value ?? "");
    }
    template = template.replace(/{{atom_data:[^}]+}}/g, "");
    return template;
}

/**
 * 
 * @returns {Promise}
 * @function awaitComponentLoad
 * @param {String} targetId The id of the target that will be listened.
 * @event
 * @description This mutation observer is required by functions that need await 
    the component orchestrator finish the proccess of building the
    component instance. by default, it stop when the first molecule 
    is appended.
 */
export async function awaitComponentLoad(targetId) {
    return Promise.race([
        new Promise((resolve, reject) => {
            const targetElement = document.getElementById(targetId);
            const organismObserver = new MutationObserver((mutations) => {
                mutations.forEach(async (mutation) => {
                    if (mutation.addedNodes.length) {
                        organismObserver.disconnect();
                        await awaitMoleculeLoad();
                    }
                });
            });
            async function awaitMoleculeLoad() {
                const moleculeObserver = new MutationObserver((mutations) => {
                    mutations.forEach(async (mutation) => {
                        if (mutation.addedNodes.length) {
                            moleculeObserver.disconnect();
                            resolve();
                            return true;
                        }
                    });
                });
                try {
                    if (targetElement.querySelector("#molecule-hydrated").children.length) {
                        resolve();
                        return true;
                    } else moleculeObserver.observe(targetElement.querySelector("#molecule-hydrated"), { childList: true });
                } catch (error) {
                    console.error(
                        "Ocorreu um erro ao tentar observar as mutações do componente. Um fallback será ativado para que a promessa seja resolvida em no máximo 5 segundos."
                    );
                    let timer = 5000;
                    const fallbackObserver = setInterval(function () {
                        console.log(timer / 1000);
                        try {
                            if (targetElement.querySelector("#molecule-hydrated").children.length) {
                                moleculeObserver.disconnect();
                                resolve();
                                console.log(`Componente carregado com sucesso.`);
                                clearInterval(fallbackObserver);
                            } else moleculeObserver.observe(targetElement.querySelector("#molecule-hydrated"), { childList: true });
                        } catch (error) {
                            console.log("Componente ainda não foi carregado.");
                        }
                        timer -= 1000;
                        if (!timer) clearInterval(fallbackObserver) && resolve();
                    }, 1000);
                }
            }
            try {
                organismObserver.observe(targetElement, { childList: true });
            } catch (error) {
                const fileName = window.location.href.split("/").filter((pageName) => pageName.includes(".html"))[0];
                console.error(`A tela ${fileName.split("?")[0]} não tem nenhum elemento com o id:  ${targetId}`);
            }
        }),
        new Promise((resolve) => setTimeout(() => resolve(), 3000)),
    ]);
}

export function taskRules() {
    const taskStatus = localStorage.codigoStatus;
    if (taskStatus != "P" && localStorage.codigoProtocolo) disableFields();
    function disableFields() {
        let allInputs = document.querySelectorAll("input:not(:read-only)");
        let allSelects = document.querySelectorAll("select");
        let allCheckboxes = document.querySelectorAll("input[type=checkbox]");
        let allButtons = document.querySelectorAll("input[type=button]");
        let allEditButtons = document.querySelectorAll("button[value=edit]");
        let actionButton = document.querySelectorAll("button#btn1010");
        let allRadios = document.querySelectorAll("input[type=radio]");
        let allTextArea = document.querySelectorAll("textarea");
        let allTagA = document.querySelectorAll("a");
        let acoes = document.querySelectorAll("#dropbtn");
        let intervalHandler = 0;
        const disableIntervalHandler = setInterval(() => {
            allInputs.forEach((e) => e.setAttribute("disabled", true));
            acoes.forEach((e) => e.setAttribute("disabled", true));
            actionButton.forEach((e) => e.setAttribute("disabled", true));
            allSelects.forEach((e) => e.setAttribute("disabled", "disabled"));
            allButtons.forEach((e) => e.setAttribute("disabled", "disabled"));
            allEditButtons.forEach((e) => e.setAttribute("disabled", "disabled"));
            allCheckboxes.forEach((e) => e.setAttribute("disabled", "disabled"));
            allRadios.forEach((e) => e.setAttribute("disabled", "disabled"));
            allTextArea.forEach((e) => e.setAttribute("readonly", true));
            allTagA.forEach((e) => {
                if (e.children[0]?.className?.includes("trash")) e.remove();
                if (e.children[0]?.className?.includes("edit")) e.remove();
            });
            document.querySelectorAll(".fa-file-signature").forEach((el) => el.parentElement.remove());
            $('input[class="ddate-picker"').prop("disabled", true);
            intervalHandler++;
            if (intervalHandler == 10) clearInterval(disableIntervalHandler);
        }, 100);
    }
}

/**
 * @param {String} componentController
 * @function cdnProvider
 * @returns {Object}
 */
function cdnProvider(componentController) {
    const cdnOrigin = isDevMode ? urlDominioFrontEnd : contextPath + "/" + contextoCDN;
    return cdnOrigin + "/components/controllers/" + componentController;
}

/**
 *
 * @param {String} componentController
 * @function importComponent
 * @async
 * @description
 * This function returns a component from CDN Provider.
 * @returns {Object}
 */
export async function importComponent(componentController) {
    const component = await import(cdnProvider(componentController));
    return component.default;
}

/**
 *
 * @param {string} action
 * @param {string | HTMLElement} node
 */
export function nodeUtils(action, node) {
    if (action === "hide") {
        if (typeof node == HTMLElement) node.classList.add("d-none");
        else getNode(node).classList.add("d-none");
    }

    if (action === "remove") {
        if (typeof node == HTMLElement) node.remove();
        else getNode(node).remove();
    }

    if (action === "show") {
        if (typeof node == HTMLElement) node.classList.remove("d-none");
        else getNode(node).classList.remove("d-none");
    }

    function getNode(id) {
        return document.getElementById(id);
    }
}
